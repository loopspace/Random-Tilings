\documentclass{article}
\usepackage[scale=.96]{geometry}
\thispagestyle{empty}
\usepackage{tikz}

\pgfmathsetmacro\tileStep{.2*(sqrt(2)-1)}

\tikzset{
  tile 1/.pic={
    \path[pic actions]
    (.2,0) arc[radius=.3,start angle=180,end angle=0]
    (.4,0) arc[radius=.1,start angle=180,end angle=0]
    (.2,1) arc[radius=.3,start angle=-180,end angle=0]
    (.4,1) arc[radius=.1,start angle=-180,end angle=0]
    (1,.2) arc[radius=.3,start angle=270,end angle=90]
    (1,.4) arc[radius=.1,start angle=270,end angle=90]
    (0,.2) arc[radius=.3,start angle=-90,end angle=90]
    (0,.4) arc[radius=.1,start angle=-90,end angle=90]
    ;
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  tile 2/.pic={
    \path[pic actions]
    (.2,0) arc[radius=.3,start angle=180,end angle=0]
    (.4,0) arc[radius=.1,start angle=180,end angle=0]
    (1,.2) arc[radius=.3,start angle=270,end angle=90]
    (1,.4) arc[radius=.1,start angle=270,end angle=90]
    (0,.2) arc[radius=.8,start angle=-90,end angle=0]
    (0,.4) arc[radius=.6,start angle=-90,end angle=0]
    (0,.6) arc[radius=.4,start angle=-90,end angle=0]
    (0,.8) arc[radius=.2,start angle=-90,end angle=0]
    ;
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  tile 3/.pic={
    \path[pic actions]
    (0,.6) arc[radius=.4,start angle=-90,end angle=0]
    (0,.8) arc[radius=.2,start angle=-90,end angle=0]
    (1,.6) arc[radius=.4,start angle=-90,end angle=-180]
    (1,.8) arc[radius=.2,start angle=-90,end angle=-180]
    (0,.4) arc[radius=.4,start angle=90,end angle=0]
    (0,.2) arc[radius=.2,start angle=90,end angle=0]
    (1,.4) arc[radius=.4,start angle=90,end angle=180]
    (1,.2) arc[radius=.2,start angle=90,end angle=180];
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  tile 4/.pic={
    \path[pic actions]
    (.2,0) |- (0,.2)
    (.4,0) |- (0,.4)
    (.6,1) |- (1,.6)
    (.8,1) |- (1,.8)
    (.2,1) |- (0,.8)
    (.4,1) |- (0,.6)
    (.6,0) |- (1,.4)
    (.8,0) |- (1,.2)
    ;
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  tile 5/.pic={
    \path[pic actions]
    (.2,0) -- (.2,.2)
    (.4,0) -- (.4,.2)
    (.6,0) -- (.6,.6)
    (.8,0) -- (.8,.6)
    (.2,1) -- (.2,.4)
    (.4,1) -- (.4,.4)
    (.6,1) -- (.6,.8)
    (.8,1) -- (.8,.8)
    (0,.2) -- (.6,.2)
    (0,.4) -- (.6,.4)
    (0,.6) -- (.2,.6)
    (0,.8) -- (.2,.8)
    (1,.2) -- (.8,.2)
    (1,.4) -- (.8,.4)
    (1,.6) -- (.4,.6)
    (1,.8) -- (.4,.8)
    ;
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  tile 6/.pic={
    \path[pic actions]
    (0,.2) -- (.2,0)
    (0,.8) -- (.2,1)
    (1,.2) -- (.8,0)
    (1,.8) -- (.8,1)
    (0,.4) -- (\tileStep,.4) -- (.4,\tileStep) -- (.4,0)
    (0,.6) -- (\tileStep,.6) -- (.4,1-\tileStep) -- (.4,1)
    (1,.4) -- (1-\tileStep,.4) -- (.6,\tileStep) -- (.6,0)
    (1,.6) -- (1-\tileStep,.6) -- (.6,1-\tileStep) -- (.6,1)
    ;
    \path[outline/.try] (0,0) rectangle (1,1);
  },
  outline/.style={},
  random tile/.code={
    \pgfmathsetmacro\randomTile{random(1,6)}%
    \tikzset{pic type=tile \randomTile}%
  }
}

\begin{document}
\vspace*{\fill}
\centering
\begin{tikzpicture}[
  scale=2.5,
]
\foreach \x in {0,...,7}
\foreach \y in {0,...,9}
\pic[transform shape,ultra thick,draw,at={(\x,\y)},random tile];
\end{tikzpicture}
\vspace*{\fill}

\end{document}
