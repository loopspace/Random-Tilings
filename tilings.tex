    -- (.1,-.5) arc[radius=.1,start angle=0,end angle=180]
    (-.3,.5) arc[radius=.3,start angle=-180,end angle=0]
    -- (.1,.5) arc[radius=.1,start angle=0,end angle=-180]
    (-.5,-.3) arc[radius=.3,start angle=-90,end angle=90]
    -- (-.5,.1) arc[radius=.1,start angle=90,end angle=-90]
    (.5,-.3) arc[radius=.3,start angle=270,end angle=90]
    -- (.5,.1) arc[radius=.1,start angle=90,end angle=270]
    ;
    \path[pic actions,fill=none]
    (-.3,-.5) arc[radius=.3,start angle=180,end angle=0]
    (-.1,-.5) arc[radius=.1,start angle=180,end angle=0]
    (-.3,.5) arc[radius=.3,start angle=-180,end angle=0]
    (-.1,.5) arc[radius=.1,start angle=-180,end angle=0]
    (.5,-.3) arc[radius=.3,start angle=270,end angle=90]

    (-.5,-.3) arc[radius=.3,start angle=-90,end angle=90]
    (-.5,-.1) arc[radius=.1,start angle=-90,end angle=90]
    ;
    \path[outline/.try] (-.5,-.5) rectangle (.5,.5);
  },
  tile 2/.pic={
    \path[pic actions,draw=none]
    (-.3,-.5) arc[radius=.3,start angle=180,end angle=0]
    -- (.1,-.5) arc[radius=.1,start angle=0,end angle=180]
    (.5,-.3) arc[radius=.3,start angle=270,end angle=90]
    -- (.5,.1) arc[radius=.1,start angle=90,end angle=270]
    (-.5,-.3) arc[radius=.8,start angle=-90,end angle=0]
    -- (.1,.5) arc[radius=.6,start angle=0,end angle=-90]
    (-.5,.1) arc[radius=.4,start angle=-90,end angle=0]
    -- (-.3,.5) arc[radius=.2,start angle=0,end angle=-90]
    ;
    \path[pic actions,fill=none]
    (-.3,-.5) arc[radius=.3,start angle=180,end angle=0]
    (-.1,-.5) arc[radius=.1,start angle=180,end angle=0]
    (.5,-.3) arc[radius=.3,start angle=270,end angle=90]
    (.5,-.1) arc[radius=.1,start angle=270,end angle=90]
    (-.5,-.3) arc[radius=.8,start angle=-90,end angle=0]
    (-.5,-.1) arc[radius=.6,start angle=-90,end angle=0]

    (-.5,.3) arc[radius=.2,start angle=-90,end angle=0]
    ;
    \path[outline/.try] (-.5,-.5) rectangle (.5,.5);
  },
  tile 3/.pic={
    \path[pic actions,draw=none]
    (-.5,.1) arc[radius=.4,start angle=-90,end angle=0]
    -- (-.3,.5) arc[radius=.2,start angle=0,end angle=-90]
    (.5,.1) arc[radius=.4,start angle=-90,end angle=-180]
    -- (.3,.5) arc[radius=.2,start angle=-180,end angle=-90]
    (-.5,-.1) arc[radius=.4,start angle=90,end angle=0]
    -- (-.3,-.5) arc[radius=.2,start angle=0,end angle=90]
    (.5,-.1) arc[radius=.4,start angle=90,end angle=180]
    -- (.3,-.5) arc[radius=.2,start angle=180,end angle=90];
    \path[pic actions,fill=none]
    (-.5,.1) arc[radius=.4,start angle=-90,end angle=0]
    (-.5,.3) arc[radius=.2,start angle=-90,end angle=0]
    (.5,.1) arc[radius=.4,start angle=-90,end angle=-180]

    (-.5,-.1) arc[radius=.4,start angle=90,end angle=0]
    (-.5,-.3) arc[radius=.2,start angle=90,end angle=0]
    (.5,-.1) arc[radius=.4,start angle=90,end angle=180]
    (.5,-.3) arc[radius=.2,start angle=90,end angle=180];

  },
  tile 4/.pic={
    \path[pic actions,draw=none]
    (-.3,-.5) |- (-.5,-.3)
    -- (-.5,-.1) -| (-.1,-.5)

    (.3,.5) |- (.5,.3)

    (-.1,.5) |- (-.5,.1)
    (.1,-.5) |- (.5,-.1)
    (.3,-.5) |- (.5,-.3)
    ;
    \path[outline/.try] (-.5,-.5) rectangle (.5,.5);
  },
  tile 5/.pic={
    \path
    (-.3,-.3) +(0,-\pgflinewidth) coordinate (a)
    (-.3,-.1) +(0,\pgflinewidth) coordinate (b)
    (.3,.3)   +(0,\pgflinewidth) coordinate (c)
    (.3,.1)   +(0,-\pgflinewidth) coordinate (d)
    (.1,-.3)  +(-\pgflinewidth,0) coordinate (e)
    (-.3,.1)  +(-\pgflinewidth,0) coordinate (f)
    (.3,-.3)  +(\pgflinewidth,0) coordinate (g)
    (-.1,.1)  +(\pgflinewidth,0) coordinate (h)
    ;
    \path[pic actions,draw=none]
    (-.3,-.5) -- (a)
    -| (-.1,-.5)
    %
    (.1,-.5) |- (d)
    -- (.3,-.5)
    %
    (-.3,.5) -- (b)
    -| (-.1,.5)
    %
    (.1,.5) |- (c)
    -- (.3,.5)
    %
    (-.5,-.3) -- (e)
    |- (-.5,-.1)
    %

    (.5,.1) -- (-.1,.1)
    (.5,.3) -- (-.1,.3)
    ;
    \path[outline/.try] (-.5,-.5) rectangle (.5,.5);
  },
  tile 6/.pic={
    \path[pic actions,draw=none]
    (-.3,-.5)
    -- (-.5,-.3) --
    (-.5,-.1) -- (-.5+\tileStep,-.1) -- (-.1,-.5+\tileStep) -- (-.1,-.5)
    %
    (-.3,.5)
    -- (-.5,.3) --
    (-.5,.1) -- (-.5+\tileStep,.1) -- (-.1,.5-\tileStep) -- (-.1,.5)
    %
    (.3,-.5)
    -- (.5,-.3) --
    (.5,-.1) -- (.5-\tileStep,-.1) -- (.1,-.5+\tileStep) -- (.1,-.5)
    %
    (.3,.5)
    -- (.5,.3) --
    (.5,.1) -- (.5-\tileStep,.1) -- (.1,.5-\tileStep) -- (.1,.5)
    %
    ;
    \path[pic actions,fill=none]
    (-.5,-.3) -- (-.3,-.5)
    (-.5,.3) -- (-.3,.5)
    (.5,-.3) -- (.3,-.5)
    (.5,.3) -- (.3,.5)
    (-.5,-.1) -- (-.5+\tileStep,-.1) -- (-.1,-.5+\tileStep) -- (-.1,-.5)

    (.5,-.1) -- (.5-\tileStep,-.1) -- (.1,-.5+\tileStep) -- (.1,-.5)
    (.5,.1) -- (.5-\tileStep,.1) -- (.1,.5-\tileStep) -- (.1,.5)
    ;
    \path[outline/.try] (-.5,-.5) rectangle (.5,.5);
  },
  outline/.style={},
  random tile/.code={
    \pgfmathsetmacro\randomTile{random(1,6)}%
    \pgfmathsetmacro\randomAngle{random(0,3)}%
    \tikzset{rotate=\randomAngle*90,pic type=tile \randomTile}%
  }
}
